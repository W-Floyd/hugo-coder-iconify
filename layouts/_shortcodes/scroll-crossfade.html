{{- /* Full-bleed scroll-through image(s). With one image it's a fixed-feeling, full-bleed image that fades in and out as you scroll. With two or more, the images dissolve directly into one another — no page background between them. A sticky stage pins the images; the crossfade opacity is driven in coder.js. Usage: {{< scroll-crossfade images="images/a.jpg" caption="A caption" >}} {{< scroll-crossfade images="images/a.jpg, images/b.jpg, images/c.jpg" >}} images required. Comma-separated list of one or more image sources (co-located page resources, site asset paths, or absolute URLs). alts optional. Comma-separated alt texts, parallel to images. caption optional. Caption shown over the clean, fade-free middle. height optional. Section height (the scroll distance). Default is one viewport per image plus one extra, e.g. 200vh for one image. fit optional. How each image fills the full-bleed stage: "cover" (default, crops to fill) or "contain" (fits the whole image, letterboxed against the page background). */ -}} {{- $imagesParam := .Get "images" | default (.Get 0) -}} {{- $srcs := slice -}} {{- range (split $imagesParam ",") -}} {{- $s := trim . " " -}} {{- with $s }}{{ $srcs = $srcs | append $s }}{{ end -}} {{- end -}} {{- $alts := slice -}} {{- with .Get "alts" -}} {{- range (split . ",") }}{{ $alts = $alts | append (trim . " ") }}{{ end -}} {{- end -}} {{- $caption := .Get "caption" -}} {{- $fit := .Get "fit" | default "cover" -}} {{- if not (in (slice "cover" "contain") $fit) -}} {{- warnf "scroll-crossfade: invalid fit %q (use \"cover\" or \"contain\"); defaulting to \"cover\"" $fit -}} {{- $fit = "cover" -}} {{- end -}} {{- $count := len $srcs -}} {{- /* One viewport per image, plus one extra for a clean hold at each end. */ -}} {{- $height := .Get "height" | default (printf "%dvh" (mul (add $count 1) 100)) -}}
{{- range $i, $src := $srcs -}} {{- $url := "" -}} {{- with $.Page.Resources.GetMatch $src -}} {{- $url = .RelPermalink -}} {{- else -}} {{- $parsed := urls.Parse $src -}} {{- $url = cond (ne $parsed.Scheme "") $src ($src | relURL) -}} {{- end -}} {{- $alt := "" -}} {{- if gt (len $alts) $i }}{{ $alt = index $alts $i }}{{ end -}} {{- end -}} {{- with $caption }}{{ end -}}
