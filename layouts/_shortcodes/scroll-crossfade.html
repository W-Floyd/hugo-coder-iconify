{{- /* One image layer in a `scroll-crossfade-group`. Renders a single full-bleed frame plus its own caption; the group pins the stage and coder.js fades each frame (and its caption with it) across its slice of the scroll. Must be nested inside a `scroll-crossfade-group` — used on its own it does nothing useful. image required. Image source (a co-located page resource, a site asset path, or an absolute URL). alt optional. Alt text for the image. caption optional. Caption shown over this image (centered by default). May contain commas or any other text. captionPosition optional. Where the caption sits. One of: top-left, top, top-right, left, center (default), right, bottom-left, bottom, bottom-right. captionSize optional. Caption text size. One of: small, medium (default), large. fit optional. How the image fills the full-bleed stage: "fill" (default, fills the frame, cropping overflow) or "contain" (fits the whole image, letterboxed against the page background). objectPosition optional. Focal point kept in view when "fill" crops the image (a CSS object-position value, e.g. "top", "left", "50% 20%"). Default center. */ -}} {{- if not .Parent -}} {{- warnf "scroll-crossfade: must be nested inside a scroll-crossfade-group" -}} {{- end -}} {{- $src := .Get "image" | default (.Get 0) -}} {{- $alt := .Get "alt" | default "" -}} {{- $caption := .Get "caption" -}} {{- $captionPositions := slice "top-left" "top" "top-right" "left" "center" "right" "bottom-left" "bottom" "bottom-right" -}} {{- $captionPosition := .Get "captionPosition" | default "center" -}} {{- if not (in $captionPositions $captionPosition) -}} {{- warnf "scroll-crossfade: invalid captionPosition %q (use one of %s); defaulting to \"center\"" $captionPosition (delimit $captionPositions ", ") -}} {{- $captionPosition = "center" -}} {{- end -}} {{- $captionSizes := slice "small" "medium" "large" -}} {{- $captionSize := .Get "captionSize" | default "medium" -}} {{- if not (in $captionSizes $captionSize) -}} {{- warnf "scroll-crossfade: invalid captionSize %q (use one of %s); defaulting to \"medium\"" $captionSize (delimit $captionSizes ", ") -}} {{- $captionSize = "medium" -}} {{- end -}} {{- $fit := .Get "fit" | default "fill" -}} {{- if not (in (slice "fill" "contain") $fit) -}} {{- warnf "scroll-crossfade: invalid fit %q (use \"fill\" or \"contain\"); defaulting to \"fill\"" $fit -}} {{- $fit = "fill" -}} {{- end -}} {{- $objectPosition := .Get "objectPosition" -}} {{- $url := "" -}} {{- with $.Page.Resources.GetMatch $src -}} {{- $url = .RelPermalink -}} {{- else -}} {{- $parsed := urls.Parse $src -}} {{- $url = cond (ne $parsed.Scheme "") $src ($src | relURL) -}} {{- end -}}
