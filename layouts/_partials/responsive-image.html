{{- /* Resolve one logical image source to a tiered set of files. Returns a dict: primary URL for the (smallest variant / the single output) srcset slice of "URL Nw" width-descriptor entries (empty when single) full URL of the largest (100%) variant (for a full-res link) hqPrimary \ hqSrcset } same three, resolved from the `-hq` companion source; empty hqFull / when no -hq files exist. For the client-side HQ toggle. Resolution priority: 1. Pre-generated -.{avif,webp} siblings (the images_build.sh size ladder) -> width-descriptor srcset, deduped by real pixel width. 2. A single pre-generated output .{avif,webp} -> plain src, no srcset. 3. Build-time Hugo generation from the source resource (AVIF) at `widths`, and (when genNative) a native-width variant for `full`. Discovery is a filename walk of the source directory, so the pipeline's output is used as-is. Returns empty primary if nothing resolves (callers may fall back, e.g. for site-asset paths or absolute URLs). A site can ship higher-quality variants alongside the stock ones by adding an `-hq` suffix to the stem (photo-hq.avif, photo-hq-1280.avif, ...). They are resolved the same way and returned as the hq* fields; the HQ/HD toggle button (see float.html / coder.js) lets a visitor switch to them at runtime. Params (dict): page required. The page (.Page) the source is relative to. src required. Source/base path, e.g. "media/photo.avif". Matching is by stem; the extension only locates the file. widths optional. Space-separated widths for the generation fallback. genNative optional. When true, the generation fallback also emits a native-width AVIF so `full` is a served file (not the source). */ -}} {{- $page := .page -}} {{- $src := .src -}} {{- $widths := .widths | default "200 400 600 800" -}} {{- $genNative := .genNative -}} {{- /* Resolve the stock source and, when it exists, an `-hq`-suffixed companion set (photo.avif -> photo-hq.avif, photo-1280.avif -> photo-hq-1280.avif). Both run through the same three-tier resolution and are returned side by side, so callers can emit both and the client-side HQ toggle (coder.js) swaps to the high-quality set at runtime where it exists. The `hq*` return fields are empty when no -hq files are present. */ -}} {{- $ext := path.Ext $src -}} {{- $hqSrc := printf "%s-hq%s" (strings.TrimSuffix $ext $src) $ext -}} {{- $results := slice -}} {{- range $cand := (slice $src $hqSrc) -}} {{- $dir := path.Dir $cand -}} {{- $stem := strings.TrimSuffix (path.Ext $cand) (path.Base $cand) -}} {{- $reStem := replaceRE `[.\[\](){}+*?^$|\\]` "\\${0}" $stem -}} {{- $pageDir := "" -}} {{- with $page.File -}}{{- $pageDir = .Dir -}}{{- end -}} {{- $scanDir := path.Join "/content" $pageDir $dir -}} {{- $set := slice -}} {{- $single := "" -}} {{- range (readDir $scanDir) -}} {{- $m := findRESubmatch (printf `^%s-([0-9]+)\.(avif|webp)$` $reStem) .Name -}} {{- if $m -}} {{- $set = $set | append (dict "name" .Name "w" (int (index (index $m 0) 1))) -}} {{- else if findRE (printf `^%s\.(avif|webp)$` $reStem) .Name -}} {{- $single = .Name -}} {{- end -}} {{- end -}} {{- $ss := slice -}} {{- $p := "" -}} {{- $f := "" -}} {{- $minW := 0 -}} {{- $maxW := 0 -}} {{- if $set -}} {{- /* 1. sized variants -> width-descriptor srcset. Use each file's real pixel width when it's a page resource (keeps a shrink-only pipeline honest); dedupe identical widths; smallest is src, largest is full. */ -}} {{- $seen := slice -}} {{- range $set -}} {{- $rel := path.Join $dir .name -}} {{- $u := "" -}}{{- $w := .w -}} {{- with $page.Resources.GetMatch $rel -}} {{- $u = .RelPermalink -}}{{- $w = .Width -}} {{- else -}} {{- $u = path.Join $page.RelPermalink $rel -}} {{- end -}} {{- if not (in $seen $w) -}} {{- $seen = $seen | append $w -}} {{- $ss = $ss | append (printf "%s %dw" $u $w) -}} {{- if or (eq $minW 0) (lt $w $minW) -}}{{- $minW = $w -}}{{- $p = $u -}}{{- end -}} {{- if gt $w $maxW -}}{{- $maxW = $w -}}{{- $f = $u -}}{{- end -}} {{- end -}} {{- end -}} {{- else if $single -}} {{- /* 2. single pre-generated output */ -}} {{- $rel := path.Join $dir $single -}} {{- with $page.Resources.GetMatch $rel -}}{{- $p = .RelPermalink -}}{{- else -}}{{- $p = path.Join $page.RelPermalink $rel -}}{{- end -}} {{- $f = $p -}} {{- else -}} {{- /* 3. build-time generation from the source resource. Only raster images can be resized; a non-processable resource (e.g. SVG) has no .Width/.Resize, so use it as-is — single output, no srcset. */ -}} {{- $img := $page.Resources.GetMatch $cand | default (resources.Get $cand) -}} {{- with $img -}} {{- if reflect.IsImageResourceProcessable . -}} {{- range (split $widths " ") -}} {{- $w := int . -}} {{- if and (gt $w 0) (le $w $img.Width) -}} {{- $r := $img.Resize (printf "%dx avif" $w) -}} {{- $ss = $ss | append (printf "%s %dw" $r.RelPermalink $w) -}} {{- if or (eq $minW 0) (lt $w $minW) -}}{{- $minW = $w -}}{{- $p = $r.RelPermalink -}}{{- end -}} {{- if gt $w $maxW -}}{{- $maxW = $w -}}{{- $f = $r.RelPermalink -}}{{- end -}} {{- end -}} {{- end -}} {{- if not $p -}}{{- $p = $img.RelPermalink -}}{{- end -}} {{- if and $genNative (gt $img.Width $maxW) -}} {{- $native := $img.Resize (printf "%dx avif" $img.Width) -}} {{- $maxW = $img.Width -}}{{- $f = $native.RelPermalink -}} {{- end -}} {{- else -}} {{- $p = .RelPermalink -}}{{- $f = .RelPermalink -}} {{- end -}} {{- end -}} {{- end -}} {{- $results = $results | append (dict "primary" $p "srcset" $ss "full" $f) -}} {{- end -}} {{- /* index 0 is the stock set, index 1 the `-hq` companion (empty fields when absent). Returned side by side for the client-side HQ toggle. */ -}} {{- $stock := index $results 0 -}} {{- $hq := index $results 1 -}} {{- return (dict "primary" $stock.primary "srcset" $stock.srcset "full" $stock.full "hqPrimary" $hq.primary "hqSrcset" $hq.srcset "hqFull" $hq.full) -}}
