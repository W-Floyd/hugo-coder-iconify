{{- /* Markdown image render hook: routes plain `![alt](src "title")` images through the same responsive-image pipeline as the {{< image >}} shortcode, so body images written in Markdown get the tiered srcset, the `data-rimg` marker, any `-hq` companion attributes, and the full-size link — i.e. the low/med/hi quality toggle works on them too. Resolution is by stem against the page's bundle/content dir (see partials/responsive-image.html). Sources that don't resolve to local files (external URLs, site-root/absolute paths) fall back to a plain with the destination used verbatim. */ -}} {{- $src := .Destination -}} {{- $alt := .Text | safeHTML -}} {{- $title := .Title -}} {{- /* Matches the shortcode default: full width on mobile, capped at the content column (860px) on desktop. px because font-relative units in `sizes` are unreliable with the preload scanner. */ -}} {{- $sizes := "(max-width: 768px) 100vw, min(100vw - 40px, 860px)" -}} {{- $r := partial "responsive-image.html" (dict "page" .Page "src" $src "widths" "200 400 600 800" "genNative" true) -}} {{- $primary := $r.primary -}} {{- $srcset := $r.srcset -}} {{- $full := $r.full -}} {{- /* Carry any `-hq` companion as data-* for the quality toggle (coder.js). */ -}} {{- $hqAttr := "" -}} {{- if $r.hqPrimary -}} {{- $hqAttr = printf ` data-hq-src=%q` $r.hqPrimary -}} {{- with $r.hqSrcset -}}{{- $hqAttr = printf `%s data-hq-srcset=%q` $hqAttr (delimit . ", ") -}}{{- end -}} {{- end -}} {{- if not $primary -}} {{- /* External/absolute/unresolved source: emit a plain image unchanged. */ -}} {{- printf `` $src $alt (cond (ne $title "") (printf ` title=%q` $title) "") | safeHTML -}} {{- else -}} {{- $img := printf `` $primary $alt (cond (gt (len $srcset) 0) (printf ` srcset=%q sizes=%q` (delimit $srcset ", ") $sizes) "") $hqAttr -}} {{- if $full -}} {{ $img | safeHTML }} {{- else -}} {{ $img | safeHTML }} {{- end -}} {{- end -}}
