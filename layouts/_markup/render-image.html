{{- /* Markdown image render hook: routes plain `![alt](src "title")` images through the same responsive-image pipeline as the {{< image >}} shortcode, so body images written in Markdown get the tiered srcset, the `data-rimg` marker, any `-hq` companion attributes, and the full-size link — i.e. the low/med/hi quality toggle works on them too. Resolution is by stem against the page's bundle/content dir (see partials/responsive-image.html). Sources that don't resolve to local files (external URLs, site-root/absolute paths) fall back to a plain with the destination used verbatim. */ -}} {{- $src := .Destination -}} {{- $alt := .Text | safeHTML -}} {{- $title := .Title -}} {{- $r := partial "responsive-img.html" (dict "page" .Page "src" $src "alt" $alt "loading" "lazy") -}} {{- if not $r.primary -}} {{- /* External/absolute/unresolved source: emit a plain image unchanged. */ -}} {{- printf `` $src $alt (cond (ne $title "") (printf ` title=%q` $title) "") | safeHTML -}} {{- else -}} {{ $r.img | safeHTML }} {{- end -}}
